[docs]def auto_label_propagators(ifd, override=False):
    """Automatically label propagators."""
    fd = copy.deepcopy(ifd)
    objs = fd.propagators
    for p in objs:
        if p.label is None or override:
            p.label = "$" + p.particle.latex_name + "$"
    return fd
