    def __cmp__(self, other):
        """Compare with another visible class, just using layers."""
        if other is None:
            return -1

        if config.getOptions().DEBUG:
            print(
                "Comparing visible classes: ",
                self.__class__,
                "->",
                self.getDepth(),
                "vs.",
                other.__class__,
                "->",
                other.getDepth(),
            )
        else:
            return cmp(self.getDepth(), other.getDepth())
