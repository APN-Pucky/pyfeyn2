[docs]def id_wrap(idd, id_type=""):
    """
    Wrap a id in the format expected by feynml.

    Args:
        idd: The id.

    Returns:
        The wrapped id.

    Examples:
        >>> id_wrap(1)
        '1'
        >>> id_wrap(-1)
        'm1'
    """
    return id_type + str(round(idd)).replace("-", "m")
