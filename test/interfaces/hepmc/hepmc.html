def rec_track(p, nchildren=-1):
    ps = [p]
    if len(p.children)==0 or (nchildren >0 and len(p.children) != nchildren):
        return ps
    for pss in p.children:
        ps = ps + rec_track(pss,nchildren)
    return ps
