    def with_point(self, p):
        self.x = float(p.x)
        self.y = float(p.y)
        return self

    def with_x(self, x):

        return self

    def with_y(self, y):
        self.y = float(y)
        return self

    def with_xy(self, x, y):
        self.x = float(x)
        self.y = float(y)
        return self

    def with_xyz(self, x, y, z):
